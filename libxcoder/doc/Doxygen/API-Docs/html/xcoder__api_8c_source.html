85 ni_device_handle_t ni_device_open(const char * p_dev)
86 {
87  ni_debug_print("ni_device_open: opening %s\n", p_dev);
88 
89 #ifdef _WIN32
90  DWORD err = 0;
91  HANDLE handle = CreateFile( p_dev,
92  GENERIC_READ | GENERIC_WRITE,
93  FILE_SHARE_READ | FILE_SHARE_WRITE,
94  NULL,
95  OPEN_EXISTING,
96  FILE_ATTRIBUTE_NORMAL,
97  NULL
98  );
99  if (INVALID_HANDLE_VALUE == handle)
100  {
101  err = GetLastError();
102  ni_debug_print("Failed to open %s, err %d \n", p_dev, err);
103  }
104  else
105  {
106  ni_debug_print("Found NVME Controller at %s \n", p_dev);
107  }
108  return handle;
109 
110 #elif __linux__
111  int err;
112  ni_device_handle_t fd = NI_INVALID_DEVICE_HANDLE;
113  fd = open(p_dev, O_RDONLY);
114  if (fd < 0)
115  {
116  return NI_RETCODE_FAILURE;
117  }
118 
119  err = fstat(fd, &g_nvme_stat);
120  if (err < 0)
121  {
122  err = NI_RETCODE_FAILURE;
123  ni_debug_print("fstat() failed on %s\n", p_dev);
124  LRETURN;
125  }
126  if (!S_ISCHR(g_nvme_stat.st_mode) && !S_ISBLK(g_nvme_stat.st_mode))
127  {
128  ni_debug_print("%s is not a block or character device\n", p_dev);
129  err = NI_RETCODE_FAILURE;
130  LRETURN;
131  }
132 
133  err = NI_RETCODE_SUCCESS;
134 
135  ni_debug_print("ni_device_open: success, fd=%d\n", fd);
136 
137  END;
138 
139  if (err < 0)
140  {
141  close(fd);
142  fd = NI_INVALID_DEVICE_HANDLE;
143  }
144 
145  return (fd);
146 #endif //__linux__
